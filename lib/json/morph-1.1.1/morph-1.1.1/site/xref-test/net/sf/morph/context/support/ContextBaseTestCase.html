305     // Test putAll()
306     public void testPutAll() {
307 
308         // Check preconditions
309 //        checkAttributeCount(0);
310         assertNull(context.get("foo"));
311         assertNull(context.get("bar"));
312         assertNull(context.get("baz"));
313         assertTrue(!context.containsKey("foo"));
314         assertTrue(!context.containsKey("bar"));
315         assertTrue(!context.containsKey("baz"));
316         assertTrue(!context.containsValue("foo value"));
317         assertTrue(!context.containsValue("bar value"));
318         assertTrue(!context.containsValue("baz value"));
319 
320         // Call putAll()
321         Map adds = new HashMap();
322         adds.put("foo", "foo value");
323         adds.put("bar", "bar value");
324         adds.put("baz", "baz value");
325         context.putAll(adds);
326 
327         // Check postconditions
328 //        checkAttributeCount(3);
329         assertEquals("foo value", (String) context.get("foo"));
330         assertEquals("bar value", (String) context.get("bar"));
331         assertEquals("baz value", (String) context.get("baz"));
332         assertTrue(context.containsKey("foo"));
333         assertTrue(context.containsKey("bar"));
334         assertTrue(context.containsKey("baz"));
335         assertTrue(context.containsValue("foo value"));
336         assertTrue(context.containsValue("bar value"));
337         assertTrue(context.containsValue("baz value"));
338 
339     }

345     // Verify the number of defined attributes
346     protected void checkAttributeCount(int expected) {
347         int actual = 0;
348         for (Iterator keys = context.keySet().iterator(); keys.hasNext(); actual++) keys.next();
349         assertEquals("Correct attribute count",
350                      expectedAttributeCount() + expected, actual);
351         if (expected == 0) {
352             assertTrue("Context should be empty", context.isEmpty());
353         } else {
354             assertFalse("Context should not be empty", context.isEmpty());
355         }
356     }
