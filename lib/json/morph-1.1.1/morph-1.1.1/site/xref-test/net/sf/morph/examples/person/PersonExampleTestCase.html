85  	protected void setUpTransformer() {
86  		// this is the overall transformer we'll use to do the graph copy
87  		graphTransformer = new SimpleDelegatingTransformer();
88  		
89  		// AddressDAO[] to String
90  		AddressDAOArrayToStringConverter addressConverter =
91  			new AddressDAOArrayToStringConverter();		
92  		// PersonDAO[] to PersonVO[]
93  		PropertyNameMappingCopier personCopier = new PersonDAOToPersonVOCopier();
94  		Map personMapping = new HashMap();
95  		personMapping.put("children", "children");
96  		personMapping.put("addresses", "primaryAddress");
97  		personMapping.put("vehicles", "vehicles");		
98  		personCopier.setMapping(personMapping);
99  		personCopier.setNestedTransformer(graphTransformer);
100 
101 		// the list of transformers that are involved in our overall graph
102 		// transformation
103 		List transformers = new ArrayList();
104 		// always put your custom transformers first
105 		transformers.add(personCopier);
106 		transformers.add(addressConverter);
107 		
108 		// then put in the default set of transformers as listed in the
109 		// SimpleDelegatingTransformer.  this makes sure all the normal conversions
110 		// you would expect from Morph are available (e.g. Integer 1 -> Long 1)
111 		transformers.add(new DefaultToBooleanConverter());
112 		transformers.add(new IdentityConverter());
113 		transformers.add(new DefaultToTextConverter());
114 		transformers.add(new TextToNumberConverter());
115 		transformers.add(new NumberConverter());
116 		transformers.add(new TextConverter());
117 		// will automatically take care of PersonDAO[] to String[]
118 		transformers.add(new ContainerCopier());
119 		// will automatically take care of VehicleDAO[] to VehicleVO[]
120 		transformers.add(new PropertyNameMatchingCopier());
121 
122 		// convert our list of transformers into an array
123 		Transformer[] transformerArray = (Transformer[]) transformers.toArray(
124 			new Transformer[transformers.size()]);
125 		graphTransformer.setComponents(transformerArray);
126 	}

128 	public void testTransformation() throws Exception {
129 		// copy the information from personDAO to personVO
130 		PersonVO johnSmithVO = new PersonVO();
131 		graphTransformer.copy(johnSmithVO, johnSmith);
132 		
133 		TestUtils.assertEquals(johnSmithVO.getName(), "John A. Smith");
134 		TestUtils.assertEquals(johnSmithVO.getPrimaryAddress(), "1 Home St");
135 		
136 		String[] childrenNames = johnSmithVO.getChildren();
137 		TestUtils.assertEquals(childrenNames[0], "Matthew");
138 		TestUtils.assertEquals(childrenNames[1], "Natalie");
139 		
140 		VehicleVO[] vehicles = johnSmithVO.getVehicles();
141 		TestUtils.assertEquals(vehicles[0].getModel(), "Taurus");
142 		TestUtils.assertEquals(vehicles[1].getModel(), "Civic");
143 	}
