106 	public void testLocaleSpecificConversions() throws Exception {
107 		double precision = 0.000001d;
108 		
109 		// text in English
110 		
111 		Double convertedDouble = (Double)
112 			getConverter().convert(Double.class, "$4,444.44", Locale.US);
113 		assertEquals(4444.44d, convertedDouble.doubleValue(), precision);
114 		
115 		convertedDouble = (Double)
116 			getConverter().convert(Double.class, "  $ 4,444.44", Locale.US);
117 		assertEquals(4444.44d, convertedDouble.doubleValue(), precision);
118 
119 		Float convertedFloat = (Float)
120 			getConverter().convert(Float.class, "35%", Locale.US);
121 		assertEquals(.35, convertedFloat.floatValue(), precision);
122 	
123 		convertedFloat = (Float)
124 			getConverter().convert(float.class, ".35%", Locale.US);
125 		assertEquals(.0035, convertedFloat.floatValue(), precision);
126 	
127 		convertedDouble = (Double)
128 			getConverter().convert(double.class, "350%", Locale.US);
129 		assertEquals(3.5, convertedDouble.floatValue(), precision);
130 
131 		convertedDouble = (Double)
132 			getConverter().convert(double.class, "3,500,000.1234", Locale.US);
133 		assertEquals(3500000.1234d, convertedDouble.doubleValue(), precision);
134 		
135 		// test in Dutch
136 		
137 		Locale dutch = new Locale("nl", "");
138 		
139 		convertedDouble = (Double)
140 			getConverter().convert(Double.class, "\u20ac4.444,44", dutch);
141 		assertEquals(4444.44d, convertedDouble.doubleValue(), precision);
142 	
143 		convertedDouble = (Double)
144 			getConverter().convert(Double.class, "\u20ac 4.444,44", dutch);
145 		assertEquals(4444.44d, convertedDouble.doubleValue(), precision);
146 
147 		convertedFloat = (Float)
148 			getConverter().convert(Float.class, "35%", dutch);
149 		assertEquals(.35, convertedFloat.floatValue(), precision);
150 	
151 		convertedFloat = (Float)
152 			getConverter().convert(float.class, ",35%", dutch);
153 		assertEquals(.0035, convertedFloat.floatValue(), precision);
154 	
155 		convertedDouble = (Double)
156 			getConverter().convert(double.class, "350%", dutch);
157 		assertEquals(3.5, convertedDouble.floatValue(), precision);
158 	
159 		convertedDouble = (Double)
160 			getConverter().convert(double.class, "3.500.000,1234", dutch);
161 		assertEquals(3500000.1234d, convertedDouble.doubleValue(), precision);
162 		
163 		Locale belgium = new Locale("nl", "BE");
164 		
165 		convertedDouble = (Double)
166 			getConverter().convert(Double.class, "4.444,44\u20ac", belgium);
167 		assertEquals(4444.44d, convertedDouble.doubleValue(), precision);
168 	
169 		convertedDouble = (Double)
170 			getConverter().convert(Double.class, "4.444,44 \u20ac", belgium);
171 		assertEquals(4444.44d, convertedDouble.doubleValue(), precision);
172 	}

174 	public void testNegativeSignHandling() throws Exception {
175 		Locale iraq = new Locale("ar", "iq");
176 		
177 		assertEquals(new Integer(-1), (Integer) getConverter().convert(Integer.class, "-1", iraq));
178 		assertEquals(new Integer(-1), (Integer) getConverter().convert(Integer.class, "1-", iraq));
179 		
180 		assertEquals(new Double(-.01), (Double) getConverter().convert(Double.class, "-1%", Locale.US));
181 		assertEquals(new Double(-.01), (Double) getConverter().convert(Double.class, "1%-", Locale.US));
