47  	public static boolean contains(Object[] array, Object object) {
48  		for (int i=0; array != null && i<array.length; i++) {
49  			if (equals(array[i], object)) {
50  				return true;
51  			}
52  		}
53  		return false;
54  	}

60  	public static boolean equals(Object object1, Object object2) {
61  		if (log.isTraceEnabled()) {
62  			log.trace("Testing for equality between "
63  				+ ObjectUtils.getObjectDescription(object1) + " and "
64  				+ ObjectUtils.getObjectDescription(object2));
65  		}
66  		// if both objects are == (incl. null) they are equal
67  		if (object1 == object2) {
68  			return true;
69  		}
70  		// if one object is null and the other is not, the two objects aren't
71  		// equal
72  		if (object1 == null || object2 == null) {
73  			return false;
74  		}
75  		if (object1 instanceof Calendar && object2 instanceof Calendar) {
76  			return equals(((Calendar) object1).getTime(), ((Calendar) object2).getTime());
77  		}
78  		if (object1 instanceof Comparable && object1.getClass() == object2.getClass()) {
79  			return ((Comparable) object1).compareTo(object2) == 0;
80  		}
81  		if (object1 instanceof Map.Entry && object2 instanceof Map.Entry) {
82  			if (object1.getClass() != object2.getClass()) {
83  				return false;
84  			}
85  			Map.Entry me1 = (Map.Entry) object1;
86  			Map.Entry me2 = (Map.Entry) object2;
87  			return equals(me1.getKey(), me2.getKey())
88  					&& equals(me1.getValue(), me2.getValue());
89  		}
90  		// if both objects are arrays
91  		if (object1.getClass().isArray() && object2.getClass().isArray()) {
92  			// if the arrays aren't of the same class, the two objects aren't
93  			// equal
94  			if (object1.getClass() != object2.getClass()) {
95  				return false;
96  			}
97  			// else, same type of array
98  			// if the arrays are different sizes, they aren't equal
99  			if (Array.getLength(object1) != Array.getLength(object2)) {
100 				return false;
101 			}
102 			// else arrays are the same size
103 			// iterate through the arrays and check if all elements are
104 			// equal
105 			for (int i=0; i<Array.getLength(object1); i++) {
106 				// if one item isn't equal to the other
107 				if (!equals(Array.get(object1, i), Array.get(object2, i))) {
108 					// the arrays aren't equal
109 					return false;
110 				}
111 			}
112 			// if we iterated through both arrays and found no items
113 			// that weren't equal to each other, the collections are
114 			// equal
115 			return true;
116 		}
117 		if (object1 instanceof Iterator && object2 instanceof Iterator) {
118 			Iterator iterator1 = (Iterator) object1;
119 			Iterator iterator2 = (Iterator) object2;
120 			while (iterator1.hasNext()) {
121 				if (!iterator2.hasNext()) {
122 					return false;
123 				}
124 				// if one item isn't equal to the other
125 				if (!equals(iterator1.next(), iterator2.next())) {
126 					// the arrays aren't equal
127 					return false;
128 				}
129 			}
130 			// if we iterated through both collections and found
131 			// no items that weren't equal to each other, the
132 			// collections are equal
133 			return !iterator2.hasNext();
134 		}
135 		if (object1 instanceof Enumeration && object2 instanceof Enumeration) {
136 			return equals(new EnumerationIterator((Enumeration) object1),
137 					new EnumerationIterator((Enumeration) object2));
138 		}
139 		if ((object1 instanceof List && object2 instanceof List) ||
140 			(object1 instanceof SortedSet && object2 instanceof SortedSet)) {
141 			// if the collections aren't of the same type, they aren't equal
142 			if (object1.getClass() != object2.getClass()) {
143 				return false;
144 			}
145 			// else same type of collection
146 			return equals(((Collection) object1).iterator(), ((Collection) object2).iterator());
147 		}
148 		if (object1 instanceof Set && object2 instanceof Set) {
149 			// if the sets aren't of the same type, they aren't equal
150 			if (object1.getClass() != object2.getClass()) {
151 				return false;
152 			}
153 			// else same type of set
154 			Set set1 = (Set) object1;
155 			Set set2 = (Set) object2;
156 			// if the sets aren't the same size, they aren't equal
157 			if (set1.size() != set2.size()) {
158 				return false;
159 			}
160 			// else sets are the same size
161 			Iterator iterator1 = set1.iterator();
162 			while (iterator1.hasNext()) {
163 				Object next = iterator1.next();
164 				if (!contains(set2, next)) {
165 					return false;
166 				}
167 			}
168 			return true;
169 		}
170 		if (object1 instanceof Map && object2 instanceof Map) {
171 			return equals(((Map) object1).entrySet(), ((Map) object2).entrySet());
172 		}
173 		if (object1.getClass() == object2.getClass() && object1 instanceof StringBuffer) {
174 			return object1.toString().equals(object2.toString());
175 		}
176 		// for primitives, use their equals methods
177 		if (object1.getClass() == object2.getClass() &&
178 			(object1 instanceof String || object1 instanceof Number ||
179 			object1 instanceof Boolean || //object1 instanceof StringBuffer ||
180 			object1 instanceof Character)) {
181 			return object1.equals(object2);
182 		}
183 		// for non-primitives, compare field-by-field
184 		return MorphEqualsBuilder.reflectionEquals(object1, object2);
185 	}

198 	/**
199 	 * Return a "not-same" instance.
200 	 * @param type
201 	 * @param o
202 	 * @return
203 	 */
204 	public static Object getDifferentInstance(Class type, Object o) {
205 		if (type == null) {
206 			throw new IllegalArgumentException("Non-null type must be specified");
207 		}
208 		if (type.isPrimitive()) {
209 			type = ClassUtils.getPrimitiveWrapper(type);
210 		}
211 		if (o != null && !type.isInstance(o)) {
212 			throw new IllegalArgumentException("Negative example object should be of type " + type);
213 		}
214 		if (type == Number.class) {
215 			type = Integer.class;
216 		}
217 		if (Number.class.isAssignableFrom(type)) {
218 			byte b = (byte) (o == null ? 0 : ((Number) o).byteValue() + 1);
219 			try {
220 				return type.getConstructor(ONE_STRING).newInstance(new Object[] { Byte.toString(b) });
221 			} catch (Exception e) {
222 				throw e instanceof RuntimeException ? (RuntimeException) e : new NestableRuntimeException(e);
223 			}
224 		}
225 		if (type == Character.class) {
226 			char c = (char) (o == null ? 0 : ((Character) o).charValue() + 1);
227 			return new Character(c);
228 		}
229 		if (type == Boolean.class) {
230 			return o != null && ((Boolean) o).booleanValue() ? Boolean.FALSE : Boolean.TRUE;
231 		}
232 		if (type.isArray()) {
233 			return Array.newInstance(type.getComponentType(), 0);
234 		}
235 		if (type == Class.class) {
236 			return o == Object.class ? Class.class : Object.class;
237 		}
238 		return ClassUtils.newInstance(convertCommonInterfaces(type));
239 	}
240 
241 	private static Class convertCommonInterfaces(Class type) {
242 		return type == Map.class ? HashMap.class : type == Set.class ? HashSet.class
243 				: type == SortedSet.class ? TreeSet.class : Collection.class
244 						.isAssignableFrom(type) ? ArrayList.class : type;
245 	}
